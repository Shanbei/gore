# npm install gore-dog -g
# add url
> gore --save "https://github.com/search?q={github}"  
> gore -s "https://github.com/search?q={github}"
# open url
> gore github gore
> auto open: https://github.com/search?q=gore
# view list
> gore --list  [,string]
> gore -l [,string]

# delect url
> gore --remove github
> gore -s github
